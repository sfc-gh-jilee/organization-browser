Page Content Area

The nav sits beside your main content in a flex layout.

HTML

<!-- Include tokens + component CSS -->
<link rel="stylesheet" href="tokens/stellar-tokens.css">
<link rel="stylesheet" href="components/globalnav/globalnav.css">

<!-- Page layout -->
<div style="display:flex; min-height:100vh">

  <!-- GlobalNav (auto-initialized via data-component) -->
  <aside
    data-component="globalnav"
    data-active="projects"
    data-user='{"name":"Danny Banks","role":"PUBLIC","hasNotifications":true}'
  ></aside>

  <main style="flex:1; padding:24px">
    <h1>Your Content Here</h1>
  </main>

</div>

<script src="components/globalnav/globalnav.js"></script>

JavaScript API

// Auto-init: any element with data-component="globalnav"
// is initialized on DOMContentLoaded.

// Manual init:
const el = document.querySelector('[data-component="globalnav"]');
const nav = new StellarGlobalNav(el);

// Listen for active item changes:
el.addEventListener('globalnav-change', (e) => {
  console.log('Active:', e.detail.activeItem);
});

// Programmatic control:
nav.setActive('monitoring');
nav.toggleCollapse();

Custom Sections

<aside
  data-component="globalnav"
  data-active="my-item"
  data-sections='[
    {
      "title": "My Section",
      "items": [
        { "id": "my-item", "label": "Custom Item", "icon": "projects" },
        { "id": "other", "label": "Another", "icon": "ai-ml" }
      ]
    }
  ]'
></aside>

Data Attributes

data-component="globalnav"   Required. Enables auto-init.
data-active="projects"       Initial active nav item ID.
data-user='{...}'            JSON: { name, role, initials?, hasNotifications? }
data-sections='[...]'        JSON: custom sections array (optional).
