CSS

.stellar-menu {
  position: absolute;
  z-index: 100000;
  min-width: max-content;
  overflow: auto;
  background-color: rgb(251, 251, 251);
  border: 1px solid rgb(189, 196, 213);
  border-radius: 6px;
  box-shadow: rgba(25, 30, 36, 0.2) 0px 4px 16px 0px;
  opacity: 0; pointer-events: none;
  transform: translateY(-4px);
  transition: opacity 0.15s ease, transform 0.15s ease;
}
.stellar-menu--open { opacity: 1; pointer-events: auto; transform: translateY(0); }

.stellar-menu__list { padding: 6px 0; min-width: 77px; max-width: 600px; outline: none; }
.stellar-menu__item { display: block; cursor: pointer; list-style: none; outline: none; }
.stellar-menu__item-label {
  display: flex; gap: 4px; padding: 8px 12px; align-items: center;
  color: rgb(30, 37, 47); cursor: pointer;
}
.stellar-menu__item:hover .stellar-menu__item-label { background-color: rgb(237, 239, 244); }
.stellar-menu__item-text {
  font-family: Inter, sans-serif; font-size: 14px; font-weight: 500; line-height: 16px;
  overflow: hidden; text-overflow: ellipsis; white-space: nowrap;
}
.stellar-menu__item-icon { width: 16px; height: 16px; flex-shrink: 0; }
.stellar-menu__divider { height: 1px; background: rgb(213, 218, 228); margin: 6px 0; }
.stellar-menu__section-header {
  padding: 12px 12px 8px; color: rgb(93, 106, 133);
  font-size: 12px; font-weight: 500; line-height: 16px;
}
.stellar-menu__item--disabled .stellar-menu__item-label {
  color: rgb(159, 171, 193); cursor: not-allowed;
}
