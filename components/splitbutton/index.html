Variants

/* Available variant classes */
stellar-splitbutton--primary     /* Blue background, white text */
stellar-splitbutton--secondary   /* White background, bordered */
stellar-splitbutton--tertiary    /* Transparent background */
stellar-splitbutton--critical    /* Red background */

/* State class */
stellar-splitbutton--disabled    /* Faded colors, cursor: not-allowed */

/* Chevron icons */
/* Primary & Critical use bold chevron (stroke-based) */
/* Secondary & Tertiary use regular chevron (fill-based) */
