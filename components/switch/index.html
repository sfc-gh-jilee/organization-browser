CSS

.stellar-switch {
  display: inline-block;
  position: relative;
  width: var(--stellar-size-md);      /* 32px */
  height: var(--stellar-size-2xs);    /* 16px */
  border-radius: var(--stellar-radius-md); /* 8px */
  background-color: var(--themed-component-form-control-background-unselected);
  transition: background-color 0.15s ease-in-out;
  cursor: pointer;
}

.stellar-switch[data-selected] {
  background-color: var(--themed-component-form-control-background-selected-default);
}

.stellar-switch__knob {
  position: absolute;
  width: var(--stellar-size-3xs);     /* 12px */
  height: var(--stellar-size-3xs);    /* 12px */
  border-radius: var(--stellar-radius-md);
  top: 50%;
  transform: translateY(-50%);
  left: var(--stellar-space-horizontal-3xs); /* 2px */
  background-color: var(--themed-component-form-control-knob-default);
  transition: left 0.1s ease-in-out;
}

.stellar-switch[data-selected] .stellar-switch__knob {
  left: calc(var(--stellar-size-md) - var(--stellar-space-horizontal-3xs) - var(--stellar-size-3xs));
}
