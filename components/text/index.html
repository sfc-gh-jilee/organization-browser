Usage

HTML

<!-- Editorial -->
<div class="stellar-editorial stellar-type-larger-editorial">Big Title</div>
<div class="stellar-editorial stellar-type-large-editorial">Title</div>

<!-- Heading -->
<h1 class="stellar-heading stellar-type-page-header">Page Header</h1>
<h2 class="stellar-heading stellar-type-sub-header">Sub Header</h2>

<!-- Paragraph -->
<p class="stellar-paragraph stellar-type-paragraph stellar-text--primary">Body text</p>
<p class="stellar-paragraph stellar-type-paragraph stellar-text--secondary">Muted text</p>
<p class="stellar-paragraph stellar-type-small-paragraph stellar-text--primary">Small text</p>
<p class="stellar-paragraph stellar-type-paragraph-bold stellar-text--primary">Bold text</p>
<p class="stellar-paragraph stellar-type-all-caps stellar-text--primary">Caps text</p>

<!-- Paragraph with Spans -->
<p class="stellar-paragraph stellar-type-paragraph stellar-text--primary">
  Regular text with <span class="stellar-span stellar-type-paragraph-bold">bold</span>
  and <span class="stellar-span stellar-text--secondary">muted</span> spans.
</p>

<!-- Label -->
<label class="stellar-label stellar-type-label stellar-text--primary">Field Label</label>
<label class="stellar-label stellar-type-label-small">Small Label</label>
<label class="stellar-label stellar-type-all-caps">Caps Label</label>

<!-- SingleLine (truncates) -->
<p class="stellar-singleline stellar-type-single-line stellar-text--primary"
   style="max-width: 200px;">Long text that truncates...</p>

<!-- Pre (monospace) -->
<pre class="stellar-pre stellar-type-paragraph stellar-text--primary">const x = 42;</pre>
<pre class="stellar-pre stellar-type-paragraph stellar-text--critical">Error: failed</pre>
